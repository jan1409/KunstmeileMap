- [ ] **Step 3.4: Implement `src/lib/map.ts`**

```ts
export function isValidCoord(lat: number, lng: number): boolean {
  if (typeof lat !== 'number' || typeof lng !== 'number') return false;
  if (!Number.isFinite(lat) || !Number.isFinite(lng)) return false;
  return lat >= -90 && lat <= 90 && lng >= -180 && lng <= 180;
}

export function clampZoom(z: number): number {
  const rounded = Math.round(z);
  return Math.min(19, Math.max(1, rounded));
}

export interface Bounds {
  south: number;
  west: number;
  north: number;
  east: number;
}

export function computeBounds(
  coords: Array<{ lat: number; lng: number }>,
): Bounds | null {
  if (coords.length === 0) return null;
  let south = coords[0].lat;
  let north = coords[0].lat;
  let west = coords[0].lng;
  let east = coords[0].lng;
  for (const c of coords) {
    if (c.lat < south) south = c.lat;
    if (c.lat > north) north = c.lat;
    if (c.lng < west) west = c.lng;
    if (c.lng > east) east = c.lng;
  }
  return { south, west, north, east };
}

const CATEGORY_PALETTE = [
  '#e57373', '#64b5f6', '#81c784', '#ffd54f',
  '#ba68c8', '#4dd0e1', '#ff8a65', '#a1887f',
];

export function colorForSlug(slug: string): string {
  let h = 0;
  for (let i = 0; i < slug.length; i++) {
    h = (h * 31 + slug.charCodeAt(i)) >>> 0;
  }
  return CATEGORY_PALETTE[h % CATEGORY_PALETTE.length];
}

export function markerColorForCategories(cats: Array<{ slug: string }>): string {
  if (cats.length === 0) return '#888';
  return colorForSlug(cats[0].slug);
}
```

```tsx
import { MapContainer, TileLayer, Marker } from 'react-leaflet';
import L from 'leaflet';
import { renderToString } from 'react-dom/server';
import 'leaflet/dist/leaflet.css';
import type { Tent } from '../lib/supabase';
import { isValidCoord, markerColorForCategories } from '../lib/map';
import { TentMarker } from './TentMarker';

interface Props {
  tents: Tent[];
  center: [number, number];
  zoom: number;
  onMarkerClick: (tent: Tent) => void;
}

export function MapView({ tents, center, zoom, onMarkerClick }: Props) {
  const placed = tents.filter(
    (t) => t.lat != null && t.lng != null && isValidCoord(t.lat, t.lng),
  );

  return (
    <MapContainer center={center} zoom={zoom} className="h-full w-full">
      <TileLayer
        attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
        url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
      />
      {placed.map((t) => {
        const color = markerColorForCategories(t.categories ?? []);
        const icon = L.divIcon({
          html: renderToString(
            <TentMarker
              displayNumber={t.display_number}
              color={color}
              ariaLabel={t.name}
            />,
          ),
          className: '',
          iconSize: [28, 28],
          iconAnchor: [14, 14],
        });
        return (
          <Marker
            key={t.id}
            position={[t.lat!, t.lng!]}
            icon={icon}
            eventHandlers={{ click: () => onMarkerClick(t) }}
          />
        );
      })}
    </MapContainer>
  );
}
```

```tsx
import { useTranslation } from 'react-i18next';
import { MapContainer, TileLayer, Marker, useMapEvents } from 'react-leaflet';
import 'leaflet/dist/leaflet.css';

interface Props {
  lat: number | null;
  lng: number | null;
  defaultCenter: [number, number];
  defaultZoom: number;
  onChange: (next: { lat: number | null; lng: number | null }) => void;
}

function MapClickHandler({ onClick }: { onClick: (lat: number, lng: number) => void }) {
  useMapEvents({
    click: (e: { latlng: { lat: number; lng: number } }) => {
      onClick(e.latlng.lat, e.latlng.lng);
    },
  });
  return null;
}

export function TentMapEditor({ lat, lng, defaultCenter, defaultZoom, onChange }: Props) {
  const { t } = useTranslation();
  const hasCoord = lat != null && lng != null;
  const markerPos: [number, number] = hasCoord ? [lat!, lng!] : defaultCenter;

  return (
    <div className="space-y-2">
      <div className="flex gap-2">
        <label className="flex-1 text-xs">
          <span className="block text-white/60">Lat</span>
          <input
            type="number"
            step="any"
            value={lat ?? ''}
            onChange={(e) => {
              const v = e.target.value === '' ? null : Number(e.target.value);
              onChange({ lat: v, lng });
            }}
            className="input mt-1"
            aria-label="Lat"
          />
        </label>
        <label className="flex-1 text-xs">
          <span className="block text-white/60">Lng</span>
          <input
            type="number"
            step="any"
            value={lng ?? ''}
            onChange={(e) => {
              const v = e.target.value === '' ? null : Number(e.target.value);
              onChange({ lat, lng: v });
            }}
            className="input mt-1"
            aria-label="Lng"
          />
        </label>
      </div>

      <div className="h-64 w-full overflow-hidden rounded border border-white/10">
        <MapContainer
          center={hasCoord ? markerPos : defaultCenter}
          zoom={defaultZoom}
          className="h-full w-full"
        >
          <TileLayer
            attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
            url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
          />
          <MapClickHandler onClick={(lt, ln) => onChange({ lat: lt, lng: ln })} />
          {hasCoord && (
            <Marker
              position={markerPos}
              draggable
              eventHandlers={{
                dragend: (e: { target: { getLatLng: () => { lat: number; lng: number } } }) => {
                  const p = e.target.getLatLng();
                  onChange({ lat: p.lat, lng: p.lng });
                },
              }}
            />
          )}
        </MapContainer>
      </div>

      <button
        type="button"
        onClick={() => onChange({ lat: null, lng: null })}
        disabled={!hasCoord}
        className="rounded bg-white/10 px-3 py-1 text-xs disabled:opacity-50"
      >
        {t('admin.tent.clear_coords') ?? 'Koordinaten löschen'}
      </button>
    </div>
  );
}
```

```ts
import { z } from 'zod';
import { isValidCoord } from './map';

export type Parser = 'csv' | 'xlsx';

export function parserForFilename(name: string): Parser | null {
  const lower = name.toLowerCase();
  if (lower.endsWith('.csv')) return 'csv';
  if (lower.endsWith('.xlsx')) return 'xlsx';
  return null;
}

export function slugify(input: string): string {
  return input
    .toLowerCase()
    .replace(/ä/g, 'ae')
    .replace(/ö/g, 'oe')
    .replace(/ü/g, 'ue')
    .replace(/ß/g, 'ss')
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '');
}

export function generateUniqueSlug(name: string, existing: string[] | Set<string>): string {
  const set = existing instanceof Set ? existing : new Set(existing);
  const base = slugify(name);
  if (!set.has(base)) return base;
  let i = 2;
  while (set.has(`${base}-${i}`)) i++;
  return `${base}-${i}`;
}

export interface RawRow {
  name?: string;
  display_number?: string;
  slug?: string;
  category_slugs?: string;
  description_de?: string;
  description_en?: string;
  address?: string;
  website_url?: string;
  instagram_url?: string;
  facebook_url?: string;
  email_public?: string;
  lat?: string;
  lng?: string;
}

export interface ParsedRow {
  name: string;
  display_number: number;
  slug: string;
  category_slugs: string[];
  description_de: string | null;
  description_en: string | null;
  address: string | null;
  website_url: string | null;
  instagram_url: string | null;
  facebook_url: string | null;
  email_public: string | null;
  lat: number | null;
  lng: number | null;
}

export interface ValidateCtx {
  knownCategorySlugs: Set<string>;
  existingDisplayNumbers: Set<number>;
  existingSlugs: Set<string>;
  rowIndex: number;
}

export interface RowResult {
  status: 'ok' | 'warning' | 'error';
  parsed?: ParsedRow;
  warnings?: string[];
  errors?: string[];
}

const UrlOrBlank = z.url().optional().or(z.literal(''));
const EmailOrBlank = z.email().optional().or(z.literal(''));

export function validateRow(row: RawRow, ctx: ValidateCtx): RowResult {
  const errors: string[] = [];
  const warnings: string[] = [];

  const name = (row.name ?? '').trim();
  if (!name) errors.push('name is required');

  const displayNumberRaw = (row.display_number ?? '').trim();
  const displayNumber = displayNumberRaw ? Number.parseInt(displayNumberRaw, 10) : NaN;
  if (!Number.isFinite(displayNumber) || displayNumber <= 0) {
    errors.push('display_number must be a positive integer');
  } else if (ctx.existingDisplayNumbers.has(displayNumber)) {
    errors.push(`display_number ${displayNumber} is duplicated`);
  }

  const slugRaw = (row.slug ?? '').trim();
  const slug = slugRaw || generateUniqueSlug(name || 'tent', ctx.existingSlugs);
  if (slugRaw && ctx.existingSlugs.has(slugRaw)) {
    errors.push(`slug "${slugRaw}" is duplicated`);
  }

  const categorySlugsRaw = (row.category_slugs ?? '').trim();
  const categorySlugs = categorySlugsRaw
    ? categorySlugsRaw.split(',').map((s) => s.trim()).filter(Boolean)
    : [];
  const unknown = categorySlugs.filter((s) => !ctx.knownCategorySlugs.has(s));
  if (unknown.length > 0) warnings.push(`unknown category slugs skipped: ${unknown.join(', ')}`);
  const validCategorySlugs = categorySlugs.filter((s) => ctx.knownCategorySlugs.has(s));

  for (const [key, raw] of [
    ['website_url', row.website_url],
    ['instagram_url', row.instagram_url],
    ['facebook_url', row.facebook_url],
  ] as const) {
    const parseResult = UrlOrBlank.safeParse(raw ?? '');
    if (!parseResult.success) errors.push(`${key} is not a valid URL`);
  }
  const emailResult = EmailOrBlank.safeParse(row.email_public ?? '');
  if (!emailResult.success) errors.push('email_public is not a valid email');

  const latRaw = (row.lat ?? '').trim();
  const lngRaw = (row.lng ?? '').trim();
  let lat: number | null = null;
  let lng: number | null = null;
  if (latRaw === '' && lngRaw === '') {
    // both blank — coords stay null, allowed
  } else if (latRaw === '' || lngRaw === '') {
    errors.push('lat and lng must both be set or both be empty');
  } else {
    lat = Number(latRaw);
    lng = Number(lngRaw);
    if (!isValidCoord(lat, lng)) errors.push('lat/lng out of range');
  }

  if (errors.length > 0) return { status: 'error', errors };

  const parsed: ParsedRow = {
    name,
    display_number: displayNumber,
    slug,
    category_slugs: validCategorySlugs,
    description_de: row.description_de?.trim() || null,
    description_en: row.description_en?.trim() || null,
    address: row.address?.trim() || null,
    website_url: row.website_url?.trim() || null,
    instagram_url: row.instagram_url?.trim() || null,
    facebook_url: row.facebook_url?.trim() || null,
    email_public: row.email_public?.trim() || null,
    lat,
    lng,
  };
  if (warnings.length > 0) return { status: 'warning', parsed, warnings };
  return { status: 'ok', parsed };
}
```

```tsx
import { useState } from 'react';
import { Link, useParams } from 'react-router-dom';
import { useTranslation } from 'react-i18next';
import Papa from 'papaparse';
import * as XLSX from 'xlsx';
import { supabase } from '../../lib/supabase';
import { useEvent } from '../../hooks/useEvent';
import { useCategories } from '../../hooks/useCategories';
import { useTents } from '../../hooks/useTents';
import {
  parserForFilename,
  validateRow,
  type RawRow,
  type RowResult,
} from '../../lib/excel';

type Step = 'upload' | 'preview' | 'done';

export default function TentImportPage() {
  const { t } = useTranslation();
  const { eventSlug } = useParams();
  const { event } = useEvent(eventSlug);
  const { categories } = useCategories(event?.id);
  const { tents: existingTents } = useTents(event?.id);

  const [step, setStep] = useState<Step>('upload');
  const [rows, setRows] = useState<RawRow[]>([]);
  const [results, setResults] = useState<RowResult[]>([]);
  const [importLog, setImportLog] = useState<string[]>([]);
  const [busy, setBusy] = useState(false);

  async function onFile(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (!file) return;
    const parser = parserForFilename(file.name);
    if (!parser) {
      alert(t('admin.import.error_unsupported_file'));
      return;
    }
    if (parser === 'csv') {
      Papa.parse<RawRow>(file, {
        header: true,
        skipEmptyLines: true,
        complete: (res) => {
          setRows(res.data);
          validateAndPreview(res.data);
        },
      });
    } else {
      const buf = await file.arrayBuffer();
      const wb = XLSX.read(buf);
      const sheet = wb.Sheets[wb.SheetNames[0]];
      const parsed = XLSX.utils.sheet_to_json<RawRow>(sheet, { defval: '' });
      // Coerce all cells to strings for uniform validation (xlsx returns numbers for number cells)
      const stringified = parsed.map((r) => {
        const out: RawRow = {};
        for (const k of Object.keys(r) as (keyof RawRow)[]) {
          const v = r[k];
          out[k] = v == null ? '' : String(v);
        }
        return out;
      });
      setRows(stringified);
      validateAndPreview(stringified);
    }
  }

  function validateAndPreview(input: RawRow[]) {
    if (!categories || !existingTents) return;
    const knownCategorySlugs = new Set(categories.map((c) => c.slug));
    const existingDisplayNumbers = new Set(
      existingTents.map((t) => t.display_number).filter((n): n is number => n != null),
    );
    const existingSlugs = new Set(existingTents.map((t) => t.slug));
    const seenDisplay = new Set<number>();
    const seenSlug = new Set<string>();

    const out: RowResult[] = input.map((row, idx) => {
      const ctx = {
        knownCategorySlugs,
        existingDisplayNumbers: new Set([...existingDisplayNumbers, ...seenDisplay]),
        existingSlugs: new Set([...existingSlugs, ...seenSlug]),
        rowIndex: idx + 2, // +2 because spreadsheets are 1-indexed and header is row 1
      };
      const res = validateRow(row, ctx);
      if (res.parsed) {
        seenDisplay.add(res.parsed.display_number);
        seenSlug.add(res.parsed.slug);
      }
      return res;
    });
    setResults(out);
    setStep('preview');
  }

  async function commitImport() {
    if (!event) return;
    setBusy(true);
    setImportLog([]);
    const log: string[] = [];

    for (let i = 0; i < results.length; i++) {
      const r = results[i];
      if (r.status === 'error' || !r.parsed) continue;
      const p = r.parsed;
      const { data: inserted, error: tentErr } = await supabase
        .from('tents')
        .insert({
          event_id: event.id,
          slug: p.slug,
          name: p.name,
          display_number: p.display_number,
          description_de: p.description_de,
          description_en: p.description_en,
          address: p.address,
          website_url: p.website_url,
          instagram_url: p.instagram_url,
          facebook_url: p.facebook_url,
          email_public: p.email_public,
          lat: p.lat,
          lng: p.lng,
        })
        .select('id')
        .single();
      if (tentErr || !inserted) {
        log.push(`❌ Row ${i + 2} (${p.slug}): ${tentErr?.message ?? 'no row returned'}`);
        continue;
      }
      if (p.category_slugs.length > 0) {
        const catIds = categories
          .filter((c) => p.category_slugs.includes(c.slug))
          .map((c) => c.id);
        const { error: joinErr } = await supabase
          .from('tent_categories')
          .insert(catIds.map((id) => ({ tent_id: inserted.id, category_id: id })));
        if (joinErr) {
          log.push(`⚠️ Row ${i + 2} (${p.slug}): tent inserted but category link failed: ${joinErr.message}`);
          continue;
        }
      }
      log.push(`✓ Row ${i + 2}: ${p.slug}`);
    }
    setImportLog(log);
    setBusy(false);
    setStep('done');
  }

  // RENDER per step
  if (step === 'upload') {
    return (
      <div className="max-w-2xl space-y-3">
        <h1 className="text-2xl font-semibold">{t('admin.import.heading')}</h1>
        <p className="text-sm text-white/60">
          {t('admin.import.help')}
        </p>
        <a className="block text-sm underline" href="/import-template.xlsx" download>
          {t('admin.import.download_template')}
        </a>
        <input type="file" accept=".csv,.xlsx" onChange={onFile} />
        <Link className="block text-sm underline" to={`/admin/events/${eventSlug}/tents`}>
          {t('common.back')}
        </Link>
      </div>
    );
  }

  if (step === 'preview') {
    const okCount = results.filter((r) => r.status === 'ok').length;
    const warnCount = results.filter((r) => r.status === 'warning').length;
    const errCount = results.filter((r) => r.status === 'error').length;
    return (
      <div className="max-w-4xl space-y-3">
        <h1 className="text-2xl font-semibold">{t('admin.import.preview_heading')}</h1>
        <p className="text-sm">
          {okCount} OK · {warnCount} warnings · {errCount} errors
        </p>
        <div className="max-h-[60vh] overflow-auto rounded border border-white/10">
          <table className="w-full text-xs">
            <thead className="bg-white/5">
              <tr><th>Row</th><th>Status</th><th>Name</th><th>#</th><th>Notes</th></tr>
            </thead>
            <tbody>
              {results.map((r, i) => (
                <tr key={i} className="border-t border-white/5">
                  <td className="px-2">{i + 2}</td>
                  <td className="px-2">{r.status === 'ok' ? '✅' : r.status === 'warning' ? '⚠️' : '❌'}</td>
                  <td className="px-2">{r.parsed?.name ?? rows[i]?.name ?? ''}</td>
                  <td className="px-2">{r.parsed?.display_number ?? rows[i]?.display_number ?? ''}</td>
                  <td className="px-2 text-white/60">
                    {(r.errors ?? []).concat(r.warnings ?? []).join('; ')}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
        <div className="flex gap-2">
          <button onClick={() => setStep('upload')} className="rounded bg-white/10 px-3 py-1">
            {t('common.back')}
          </button>
          <button
            onClick={commitImport}
            disabled={busy || okCount + warnCount === 0}
            className="rounded bg-white/20 px-3 py-1 disabled:opacity-50"
          >
            {busy ? '…' : t('admin.import.commit_button', { count: okCount + warnCount })}
          </button>
        </div>
      </div>
    );
  }

  // done
  return (
    <div className="max-w-2xl space-y-3">
      <h1 className="text-2xl font-semibold">{t('admin.import.done_heading')}</h1>
      <pre className="max-h-[60vh] overflow-auto rounded bg-black/40 p-2 text-xs">
        {importLog.join('\n')}
      </pre>
      <Link to={`/admin/events/${eventSlug}/tents`} className="block text-sm underline">
        {t('common.back')}
      </Link>
    </div>
  );
}
```
